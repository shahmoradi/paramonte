38 import numpy as np
39 whole = np.round(val)
40 # abs(residual) < .5 always holds.
41 residual = val - whole
42 urand = np.random.uniform(0, 1, (np.size(val, axis = 0), np.size(val, axis = 1)))
43 mask = 1 - np.abs(residual) < urand
44 mask1 = np.logical_and(mask, residual < 0)
45 mask2 = np.logical_and(mask, residual >= 0)
46 whole[np.where(mask1)] = whole[np.where(mask1)] - 1
47 whole[np.where(mask2)] = whole[np.where(mask2)] + 1
48 return whole
