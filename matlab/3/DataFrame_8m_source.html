159 %> \brief
160 %> Generate and return a natural logarithmically-spaced
161 %> range of indices from the row indices of the input
162 %> dataframe ``dfref`` to the parent object.
163 %>
164 %> \details
165 %> This method is a convenience wrapper
166 %> around function ``pm.array.logrange()``.
167 %>
168 %> \warning
169 %> Beware of the different order of the input arguments
170 %> between this method and ``pm.array.logrange()``.
171 %>
172 %> \param[in] count : The input scalar MATLAB whole-number (integer)
173 %> representing the maximum size of the output range.<br>
174 %> Due to rounding operation involved in creating the
175 %> output range, it is impossible to prespecify the
176 %> output range size, only the maximum.<br>
177 %> (**optional**, default = ``1000``)
178 %>
179 %> \param[in] start : The input scalar MATLAB whole-number (integer)
180 %> representing the starting point of the output range.<br>
181 %> It must be a number in the range ``[1, size(self.dfref, 1)]``.
182 %> Otherwise, the value ``max(1, min(start, self.nrow()))`` will be used.<br>
183 %> (**optional**, default = ``1``)
184 %>
185 %> \param[in] stop : The input scalar MATLAB whole-number (integer)
186 %> representing the stopping point of the output range.<br>
187 %> It must be a number in the range ``[1, size(self.dfref, 1)]``.
188 %> Otherwise, the value ``max(start, min(stop, self.nrow()))`` will be used.<br>
189 %> (**optional**, default = ``1``)
190 %>
191 %> \return
192 %> `indices` : The output vector of MATLAB real values containing
193 %> the set of naturally logarithmically-spaced integer
194 %> values in the specified input range.
195 %>
196 %> \interface{rowslog}
197 %> \code{.m}
198 %>
199 %> indices = pm.data.DataFrame()
200 %> indices = pm.data.DataFrame([])
201 %>
202 %> indices = pm.data.DataFrame([], [])
203 %> indices = pm.data.DataFrame([], start)
204 %> indices = pm.data.DataFrame(count, [])
205 %> indices = pm.data.DataFrame(count, start)
206 %>
207 %> indices = pm.data.DataFrame([], [], [])
208 %> indices = pm.data.DataFrame(count, [], [])
209 %> indices = pm.data.DataFrame([], start, [])
210 %> indices = pm.data.DataFrame([], [], stop)
211 %> indices = pm.data.DataFrame([], start, stop)
212 %> indices = pm.data.DataFrame(count, [], stop)
213 %> indices = pm.data.DataFrame(count, start, [])
214 %> indices = pm.data.DataFrame(count, start, stop)
215 %>
216 %> \endcode
217 %> \final{rowslog}
218 %>
219 %> \author
220 %> \JoshuaOsborne, May 21 2024, 5:10 PM, University of Texas at Arlington<br>
221 %> \FatemehBagheri, May 20 2024, 1:25 PM, NASA Goddard Space Flight Center, Washington, D.C.<br>
222 %> \AmirShahmoradi, May 16 2016, 9:03 AM, Oden Institute for Computational Engineering and Sciences (ICES), UT Austin<br>
223 function indices = rowslog(self, count, start, stop)
224 if nargin < 4
225 stop = [];
226 end
227 if nargin < 3
228 start = [];
229 end
230 if nargin < 2
231 count = [];
232 end
233 if isempty(stop)
234 stop = self.nrow();
235 end
236 if isempty(start)
237 start = 1;
238 end
239 if isempty(count)
240 count = 1000;
241 end
242 indices = pm.array.logrange(max(1, min(start, self.nrow())), max(start, min(stop, self.nrow())), count);
243 end
