1function colornames_view(palette,order)
2% View the COLORNAMES palettes in an interactive figure. Sort colors by name/color space.
3%
4% (c) 2014-2019 Stephen Cobeldick
5%
6%%% Syntax:
7% colornames_view
8% colornames_view(palette)
9% colornames_view(palette,sortorder)
10%
11% Create a figure displaying all of the colors from any palette supported
12% by the function COLORNAMES. The palette and sort order can be selected
13% by drop-down menu or by optional inputs. The colors can be sorted:
14% - alphabetically by color name.
15% - by color space parameters: Lab, LCh, XYZ, YUV, HSV, or RGB.
16%
17% The figure is resizeable (the colors flow to fit the axes) with a vertical
18% scrollbar provided to view all of the colors (if required). A rudimentary
19% zoom ability has also been implemented... but this is rather experimental!
20%
21% Note: Requires the function COLORNAMES and its associated MAT file (FEX 48155).
22%
23%% Input and Output Arguments %%
24%
25%%% Inputs (all inputs are optional):
26% palette = CharRowVector, the name of a palette supported by COLORNAMES.
27% sortorder = CharRowVector, the color space sort parameters in the desired order,
28% eg: 'RGB', 'BRG', 'GRB', 'Lab', 'abL', etc, OR 'Alphabetical'.
29%
30%%% Outputs:
31% none
32%
33% See also COLORNAMES COLORNAMES_CUBE COLORNAMES_DELTAE MAXDISTCOLOR COLORMAP
34
35%% Figure Parameters %%
36%
37persistent fgh axh slh pah srh txh txs rgb coh prv
38%
39% Text lightness threshold:
40thr = 0.54;
41%
42% Text margin, uicontrol and axes gap (pixels):
43mrg = 5;
44gap = 4;
45sid = 20;
46%
47% Slider position:
48yid = 0;
49ymx = 0;
50%
51% Handle of outlined text:
52prv = [];
53%
54pmt = 'Click on a colorname...';
55%
56%% Get COLORNAME Palettes %%
57%
58[fnm,csf] = colornames();
59%
60if nargin<1
61 pnm = fnm{1+rem(round(now*1e7),numel(fnm))};
62else
63 assert(ischar(palette)&&isrow(palette),'First input <palette> must be a 1xN char.')
64 idz = strcmpi(palette,fnm);
65 assert(any(idz),'Palette ''%s'' is not supported. Call COLORNAMES() to list all palettes.',palette)
66 pnm = fnm{idz};
67end
68%
69%% Color Sorting List %%
70%
71% List of color spaces for sorting:
72lst = {'Alphabetical','Lab','XYZ','LCh','YUV','HSV','RGB'};
73[tsl,idt] = cellfun(@sort,lst,'UniformOutput',false);
74% Get every permutation of the color spaces:
75trs = cellfun(@(s)perms(s(end:-1:1)),lst(2:end),'UniformOutput',false);
76trs = [lst(1);cellstr(vertcat(trs{:}))];
77%
78if nargin<2
79 srt = lst{1};
80else
81 assert(ischar(order)&&isrow(order),'Second input <sort> must be a 1xN char.')
82 idz = strcmpi(order,trs);
83 assert(any(idz),'Second input must be one of:%s\b.',sprintf(' %s,',trs{:}))
84 srt = trs{idz};
85end
86%
87% Intial color sorting index:
88idx = 1:numel(colornames(pnm));
89%
90%% Create a New Figure %%
91%
92if isempty(fgh)||~ishghandle(fgh)
93 % Figure with zoom and pan functions:
94 fgh = figure('HandleVisibility','callback', 'IntegerHandle','off',...
95 'NumberTitle','off', 'Name','ColorNames View', 'Color','white',...
96 'Toolbar','figure', 'Units','pixels', 'Tag',mfilename, 'Visible','on');
97 %
98 fgp = get(fgh,'Position');
99 inh = uicontrol(fgh, 'Units','pixels', 'Style','text', 'HitTest','off',...
100 'Visible','on', 'String','Initializing the figure... please wait.');
101 inx = get(inh,'Extent');
102 set(inh,'Position',[fgp(3:4)/2-inx(3:4)/2,inx(3:4)])
103 %
104 % Axes and scrolling slider:
105 slh = uicontrol('Parent',fgh, 'Style','slider', 'Visible','off',...
106 'Enable','on', 'Value',1, 'Min',0, 'Max',1,...
107 'FontUnits','pixels', 'Units','pixels', 'Callback',@cnvSldClBk);
108 axh = axes('Parent',fgh, 'Visible','off', 'Units','pixels',...
109 'YDir','reverse', 'XTick',[], 'YTick',[], 'XLim',[0,1], 'YLim',[0,1]);
110 % Palette and color sorting method drop-down menus:
111 pah = uicontrol('Parent',fgh, 'Style','popupmenu', 'String',fnm,...
112 'ToolTip','Color Scheme', 'Units','pixels',...
113 'Visible','off', 'Callback',@cnvPalClBk);
114 srh = uicontrol('Parent',fgh, 'Style','popupmenu', 'String',trs,...
115 'ToolTip','Sort Colors', 'Units','pixels',...
116 'Visible','off', 'Callback',@cnvSrtClBk);
117 coh = uicontrol('Parent',fgh, 'Style','edit', 'String',pmt,...
118 'ToolTip','RGB Value', 'Units', 'pixels', 'Visible','off',...
119 'HorizontalAlignment','left', 'Enable','inactive');
120else
121 set(coh,'String',pmt)
122end
123set(pah,'Value',find(strcmpi(pnm,fnm)));
124set(srh,'Value',find(strcmpi(srt,trs)));
125%
126fgo = get(fgh, 'Pointer');
127set(fgh, 'Pointer','watch')
128drawnow()
129%
130%% Callback Functions %%
131%
132 function cnvPalClBk(h,~) % Palette Menu CallBack
133 % Select a new palette.
134 pnm = fnm{get(h,'Value')};
135 set(slh, 'Value',1)
136 set(coh, 'String',pmt)
137 set(fgh, 'Pointer','watch')
138 drawnow()
139 cnvTxtDraw()
140 cnvSortBy()
141 cnvResize()
142 set(fgh, 'Pointer',fgo)
143 end
144%
145 function cnvSrtClBk(h,~) % Sort-Order Menu CallBack
146 % Select the color sorting method.
147 srt = trs{get(h,'Value')};
148 set(fgh, 'Pointer','watch')
149 drawnow()
150 cnvSortBy()
151 cnvResize()
152 set(fgh, 'Pointer',fgo)
153 end
154%
155 function cnvSldClBk(h,~) % Slider CallBack
156 % Scroll the axes by changing the axes limits.
157 tmp = diff(get(axh,'Ylim'));
158 set(axh, 'Ylim',[0,tmp] + (ymx-tmp)*(1-get(h,'Value')));
159 end
160%
161 function cnvZoomClBk(~,~) % Zoom CallBack
162 % Change the font and margin sizes.
163 tmp = diff(get(axh,'Ylim'));
164 set(txh, 'FontSize',txs/tmp);
165 set(txh, 'Margin',mrg/tmp);
166 end
167%
168 function cnvPanClBk(~,~) % Pan CallBack
169 % Move the scroll-bar to match panning of the axes.
170 tmp = get(axh,'Ylim');
171 set(slh, 'Value',max(0,min(1,1-tmp(1)/(ymx-diff(tmp)))))
172 end
173%
174%% Color Sorting %%
175%
176 function cnvSortBy()
177 [tmp,ids] = sort(srt);
178 idc = strcmp(tmp,tsl);
179 ids = ids(idt{idc});
180 switch lst{idc}
181 case 'Alphabetical'
182 idx = 1:numel(txh);
183 return
184 case 'RGB'
185 mat = rgb;
186 case 'HSV'
187 mat = csf.rgb2hsv(rgb);
188 case 'XYZ'
189 mat = csf.rgb2xyz(rgb);
190 case 'Lab'
191 mat = csf.xyz2lab(csf.rgb2xyz(rgb));
192 case 'LCh'
193 mat = csf.lab2lch(csf.xyz2lab(csf.rgb2xyz(rgb)));
194 case 'YUV' % BT.709
195 mat = csf.invgamma(rgb) * [...
196 +0.2126, -0.19991, +0.61500;...
197 +0.7152, -0.33609, -0.55861;...
198 +0.0722, +0.43600, -0.05639];
199 otherwise
200 error('Colorspace "%s" is not supported. How did this happen?',srt)
201 end
202 [~,idx] = sortrows(mat,ids);
203 end
204%
205%% Re/Draw Text Strings %%
206%
207txf = @(s,b,c)text('Parent',axh, 'String',s, 'BackgroundColor',b,...
208 'Color',c,'Margin',mrg, 'Units','data', 'Interpreter','none',...
209 'VerticalAlignment','bottom', 'HorizontalAlignment','right',...
210 'Clipping','on', 'ButtonDownFcn',{@txtClBk,s,b},'LineWidth',3);
211%
212 function txtClBk(hnd,~,s,bgd)
213 try %#ok<TRYNC>
214 set(prv, 'EdgeColor','none')
215 end % setting one object is much faster than setting all objects.
216 prv = hnd;
217 %
218 hxs = sprintf('%02X',round(bgd*255));
219 dcs = sprintf(',%.5f',bgd);
220 set(coh, 'String',sprintf('#%s [%s] %s',hxs,dcs(2:end),s));
221 set(hnd, 'EdgeColor',max(0,min(1,1-bgd)))
222 uistack(hnd,'top')
223 end
224%
225 function cnvTxtDraw()
226 % Delete any existing colors:
227 delete(txh(ishghandle(txh)))
228 % Get new colors:
229 [cnc,rgb] = colornames(pnm);
230 % Calculate the text color:
231 baw = (rgb*[0.298936;0.587043;0.114021])<thr;
232 % Draw new colors in the axes:
233 txh = cellfun(txf,cnc,num2cell(rgb,2),num2cell(baw(:,[1,1,1]),2),'Uni',0);
234 txh = reshape([txh{:}],[],1);
235 txs = get(txh(1),'FontSize');
236 prv = txh(1);
237 end
238%
239%% Resize the Axes and UIControls, Move the Colors %%
240%
241 function cnvResize(~,~)
242 %
243 zoom(fgh,'out')
244 %
245 if nargin
246 set(fgh, 'Pointer','watch')
247 drawnow()
248 end
249 %
250 ecv = get(prv, 'EdgeColor');
251 set(prv, 'EdgeColor','none');
252 %
253 set(axh, 'Ylim',[0,1])
254 set(txh, 'Units','pixels', 'FontSize',txs, 'Margin',mrg)
255 txe = cell2mat(get(txh(:),'Extent'));
256 top = get(slh,'FontSize')*2;
257 fgp = get(fgh,'Position'); % [left bottom width height]
258 hgt = round(fgp(4)-3*gap-top);
259 wid = fgp(3)-3*gap-sid;
260 pos = [gap,gap,wid,hgt];
261 %
262 % Calculate color lengths from text and margins:
263 txw = 2*mrg+txe(idx,3);
264 txc = cumsum(txw);
265 % Preallocate position array:
266 txm = mean(txw);
267 out = zeros(ceil(1.1*[txc(end)/pos(3),pos(3)/txm]));
268 % Split colors into lines that fit the axes width:
269 idb = 1;
270 idr = 0;
271 tmp = 0;
272 while idb<=numel(txc)
273 idr = idr+1;
274 idp = max([idb,find((txc-tmp)<=pos(3),1,'last')]);
275 out(idr,1:1+idp-idb) = txc(idb:idp)-tmp;
276 tmp = txc(idp);
277 idb = idp+1;
278 end
279 % Calculate X and Y positions for each color:
280 [~,txy,txx] = find(out.');
281 yid = txy(end);
282 txy = txy*(2*mrg+max(txe(idx,4)));
283 ymx = txy(end)/pos(4);
284 %
285 % Resize the scrollbar, adjust scroll steps:
286 nwp = [2*gap+wid,gap,sid,hgt];
287 if ymx>1
288 set(slh, 'Position',nwp, 'Enable','on', 'Value',1,...
289 'SliderStep',max(0,min(1,[0.5,2]/(yid*(ymx-1)/ymx))))
290 else
291 set(slh, 'Position',nwp, 'Enable','off')
292 end
293 % Resize the axes and drop-down menus:
294 set(axh, 'Position',pos)
295 uiw = (fgp(3)-gap)/4-gap;
296 txw = 2*uiw+gap;
297 lhs = gap+(0:2)*(uiw+gap);
298 bot = fgp(4)-top-gap;
299 set(pah, 'Position',[lhs(1),bot,uiw,top])
300 set(srh, 'Position',[lhs(2),bot,uiw,top])
301 set(coh, 'Position',[lhs(3),bot,txw,top])
302 % Move text strings to the correct positions:
303 arrayfun(@(h,x,y)set(h,'Position',[x,y]),txh(idx),txx-mrg,pos(4)-txy+mrg);
304 set(txh, 'Units','data')
305 %
306 set(prv, 'EdgeColor',ecv);
307 %
308 if nargin
309 set(fgh, 'Pointer',fgo)
310 end
311 drawnow()
312 end
313%
314%% Initialize the Figure %%
315%
316cnvTxtDraw()
317cnvSortBy()
318cnvResize()
319set([pah,srh,coh,slh], 'Visible','on')
320set(fgh, 'Pointer',fgo, 'ResizeFcn',@cnvResize)
321set(zoom(fgh), 'ActionPostCallback',@cnvZoomClBk);
322set(pan(fgh), 'ActionPostCallback',@cnvPanClBk);
323try %#ok<TRYNC>
324 delete(inh)
325end
326%
327end
