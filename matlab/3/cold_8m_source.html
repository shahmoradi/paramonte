67function cmap = cold(nell)
68 if nargin < 1
69 nell = [];
70 end
71 if isempty(nell)
72 nell = size(get(gcf, 'colormap'), 1);
73 end
74 n = fix(3 / 8 * nell);
75 r = [zeros(2 * n, 1); (1 : nell - 2 * n)' / (nell - 2 * n)];
76 g = [zeros(n, 1); (1 : n)' / n; ones(nell - 2 * n, 1)];
77 b = [(1 : n)' / n; ones(nell - n, 1)];
78 cmap = [r, g, b];
79end
