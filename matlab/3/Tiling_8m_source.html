161 fullSize = get(0, 'ScreenSize');
162 maxSize = fullSize;

168 newHeight = figSize(4) * min(maxScale(2), size(self.subplot, 1));
169 figStart = [(fullSize(3) - newWidth) / 2, (fullSize(4) - newHeight) / 2];
