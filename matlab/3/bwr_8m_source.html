1%> \brief
2%> Return a blue-white-red colormap matrix
3%> of shape ``(nell, 3)`` containing the RGB values with
4%> white corresponding to the CAXIS value closest to zero.
5%>
6%> \note
7%> This colormap is most useful for images and surface plots
8%> with positive and negative values.
9%>
10%> \param[in] nell : The input scalar MATLAB integer representing
11%> the number of rows of the output colormap matrix.
12%> (**optional**, default = ``size(get(gcf, 'colormap'), 1)``)
13%>
14%> \param[in] clim : The input vector length ``2`` of MATLAB doubles
15%> representing the color limits of the output colormap.
16%> (**optional**, default = ``get(gca, 'CLim')``)
17%>
18%> \return
19%> `cmap` : The output blue-white-red colormap matrix
20%> of shape ``[nell, 3]`` containing a **cold** colormap.
21%>
22%> \interface{bwr}
23%> \code{.m}
24%>
25%> cmap = pm.vis.cmap.cold()
26%> cmap = pm.vis.cmap.cold([])
27%> cmap = pm.vis.cmap.cold(nell)
28%>
29%> \endcode
30%>
31%>
32%> \example{bwr}
33%>
34%> figure;
35%> imagesc(peaks(500));
36%> colormap(pm.vis.cmap.bwr(256));
37%> colorbar;
38%>
39%> figure;
40%> imagesc(peaks(500), [0, 8]);
41%> colormap(pm.vis.cmap.bwr());
42%> colorbar;
43%>
44%> figure;
45%> imagesc(peaks(250), [-6, 0]);
46%> colormap(pm.vis.cmap.bwr());
47%> colorbar;

89function newmap = bwr(nell, clim)

91 if nargin < 1 || isempty(nell)
92 nell = size(get(gcf,'colormap'), 1);
93 end
94 bottom = [0, 0, 0.5];
95 botmiddle = [0, 0.5, 1];
96 topmiddle = [1, 0, 0];
97 middle = [1, 1, 1];
98 top = [0.5, 0, 0];
99
100 % Find middle.
101
102 try
103 lims = get(gca, 'CLim');
104 catch
105 lims = clim;
106 end
107
108 % Find ratio of negative to positive.
109
110 if (lims(1) < 0) & (lims(2) > 0)
111
112 % It has both negative and positive
113 % Find ratio of negative to positive
114 ratio = abs(lims(1)) / (abs(lims(1)) + lims(2));
115 neglen = round(nell*ratio);
116 poslen = nell - neglen;
117
118 % Just negative
119
120 new = [bottom; botmiddle; middle];
121 len = length(new);
122 oldsteps = linspace(0, 1, len);
123 newsteps = linspace(0, 1, neglen);
124 newmap1 = zeros(neglen, 3);
125
126 for i = 1 : 3
127 % Interpolate over RGB spaces of colormap
128 newmap1(:,i) = min(max(interp1(oldsteps, new(:,i), newsteps)', 0), 1);
129 end
130
131 % Just positive.
132
133 new = [middle; topmiddle; top];
134 len = length(new);
135 oldsteps = linspace(0, 1, len);
136 newsteps = linspace(0, 1, poslen);
137 newmap = zeros(poslen, 3);
138
139 for i = 1 : 3
140 % Interpolate over RGB spaces of colormap
141 newmap(:,i) = min(max(interp1(oldsteps, new(:,i), newsteps)', 0), 1);
142 end
143
144 % And put them together.
145
146 newmap = [newmap1; newmap];
147
148 elseif lims(1) >= 0
149
150 % Just positive.
151
152 new = [middle; topmiddle; top];
153 len = length(new);
154 oldsteps = linspace(0, 1, len);
155 newsteps = linspace(0, 1, nell);
156 newmap = zeros(nell, 3);
157
158 for i = 1 : 3
159 % Interpolate over RGB spaces of colormap
160 newmap(:,i) = min(max(interp1(oldsteps, new(:,i), newsteps)', 0), 1);
161 end
162
163 else
164
165 % Just negative.
166
167 new = [bottom; botmiddle; middle];
168 len = length(new);
169 oldsteps = linspace(0, 1, len);
170 newsteps = linspace(0, 1, nell);
171 newmap = zeros(nell, 3);
172
173 for i=1:3
174 % Interpolate over RGB spaces of colormap
175 newmap(:,i) = min(max(interp1(oldsteps, new(:,i), newsteps)', 0), 1);
176 end
177
178 end
