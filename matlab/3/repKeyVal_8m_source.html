38function hashnew = repKeyVal(key, val, hashmap)
39 hashnew = hashmap;
40 vararginLen = length(hashnew);
41 keyString = string(key);
42 failed = true;
43 for i = 1 : vararginLen - 1
44 try
45 if strcmpi(string(hashnew{i}), keyString)
46 hashnew{i + 1} = val;
47 failed = false;
48 break;
49 end
50 catch
51 continue
52 end
53 end
54 if failed
55 hashnew = {hashnew{:}, key, val};
56 end
57end
