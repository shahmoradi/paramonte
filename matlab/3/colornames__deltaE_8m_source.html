1function colornames_deltaE(palette,rgb)
2% Create a figure comparing the color difference (deltaE) calculations used in COLORNAMES.
3%
4% (c) 2014-2019 Stephen Cobeldick
5%
6%%% Syntax:
7% colornames_deltaE(palette,rgb)
8%
9% Creates a figure showing the supplied colormap as horizontal color bands,
10% overlaid with columns of the closest named colors from the selected
11% palette. Each column shows one color difference (deltaE) calculation.
12%
13% Note: Requires the function COLORNAMES and its associated MAT file (FEX 48155).
14%
15% For more information on color difference concepts and formulae:
16% https://en.wikipedia.org/wiki/Color_difference
17% http://www.colorwiki.com/wiki/Delta_E:_The_Color_Difference
18%
19%% Examples %%
20%
21% colornames_deltaE('html4',jet(18))
22%
23% colornames_deltaE('x11',summer(18))
24%
25% colornames_deltaE('matlab',jet(18))
26%
27%% Input and Output Arguments %%
28%
29%%% Inputs (all inputs are optional):
30% palette = CharRowVector, the name of a palette supported by COLORNAMES.
31% rgb = Numeric Array, size Nx3, each row is an RGB triple (0<=rgb<=1).
32%
33%%% Outputs:
34% none
35%
36% See also COLORNAMES COLORNAMES_CUBE COLORNAMES_VIEW MAXDISTCOLOR COLORMAP
37
38% Get list of palettes and color difference metrics:
39[fnm,~,dtE] = colornames();
40%
41% Define default values:
42if nargin<2
43 N = 16;
44 map = get(0,'defaultFigureColormap');
45 rgb = interp1(linspace(1,N,size(map,1)),map,1:N);
46end
47if nargin<1
48 palette = fnm{1+rem(round(now*1e7),numel(fnm))};
49end
50%
51persistent fgh axh txh
52%
53% Text lightness threshold:
54thr = 0.54;
55%
56if isempty(fgh)||~ishghandle(fgh)
57 fgh = figure('HandleVisibility','callback', 'IntegerHandle','off',...
58 'NumberTitle','off', 'Name',mfilename, 'Color','white', 'Toolbar','none');
59 axh = axes('Parent',fgh, 'Visible','off', 'XTick',[], 'YTick',[],...
60 'Units','normalized', 'Position',[0,0,1,1]);
61else
62 try %#ok<TRYNC>
63 cla(axh)
64 end
65end
66%
67assert(ischar(palette)&&isrow(palette),'First input <palette> must be a 1xN char.')
68idz = strcmpi(palette,fnm);
69assert(any(idz),'Palette ''%s'' is not supported. Call COLORNAMES() to list all palettes.',palette)
70set(fgh,'Name',sprintf('%s (palette = ''%s'')',mfilename,fnm{idz}))
71%
72assert(ismatrix(rgb)&&size(rgb,2)==3&&isreal(rgb)&&all(0<=rgb(:)&rgb(:)<=1),...
73 'Second input argument can be a colormap of RGB values (size Nx3).')
74colormap(axh,rgb);
75%
76N = size(rgb,1);
77x = [0;0;1;1];
78y = [0;1;1;0];
79X = repmat(x,1,N);
80Y = bsxfun(@plus,y,N-1:-1:0);%0:N-1);
81patch(X,Y,1:N, 'Parent',axh, 'EdgeColor','none', 'FaceColor','flat', 'CDataMapping','direct');
82%
83dEn = numel(dtE);
84[cnc,RGB] = cellfun(@(t)colornames(palette,rgb,t),dtE, 'uni',false);
85BAW = cellfun(@(c)(c*[0.298936;0.587043;0.114021])<thr,RGB, 'uni',false);
86%
87tmp = @(s,n) text((2*n-1)*ones(1,N)/(2*dEn), mean(Y.',2), zeros(1,N),...
88 s, 'Parent',axh, 'HorizontalAlignment','center');
89txh = cellfun(tmp, cnc, num2cell(1:dEn), 'uni',false);
90tmp = @(h,c,b) set(h(:), {'BackgroundColor'},num2cell(c,2), {'Color'},num2cell(b(:,[1,1,1]),2));
91cellfun(tmp, txh, RGB, BAW)
92%
93set(axh,'YLim',[0,N+1]);
94text((1:2:2*dEn)/(2*dEn), N+ones(1,dEn)/2, zeros(1,dEn), dtE(:),...
95 'Parent',axh, 'HorizontalAlignment','center', 'Color','black');
96%
97drawnow()
98%
99end
