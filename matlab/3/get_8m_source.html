1%> SUBAXIS Create axes in tiled positions. (just like subplot)
2%> Usage:
3%> h=getAxes(rows,cols,cellno[,settings])
4%> h=getAxes(rows,cols,cellx,celly[,settings])
5%> h=getAxes(rows,cols,cellx,celly,spanx,spany[,settings])
6%>
7%> SETTINGS: Spacing,SpacingHoriz,SpacingVert
8%> Padding,PaddingRight,PaddingLeft,PaddingTop,PaddingBottom
9%> Margin,MarginRight,MarginLeft,MarginTop,MarginBottom
10%> Holdaxis
11%>
12%> all units are relative (i.e. from 0 to 1)
13%>
14%> Abbreviations of parameters can be used.. (Eg MR instead of MarginRight)
15%> (holdaxis means that it wont delete any axes below.)
16%>
17%>
18%>
19%> \example{getAxes}
20%>
21%> >> getAxes(2,1,1,'SpacingVert',0,'MR',0);
22%> >> imagesc(magic(3))
23%> >> getAxes(2,'p',.02);
24%> >> imagesc(magic(4))
25%>

27function h = getAxes(varargin)
28 f=gcf;
29 UserDataArgsOK=0;
30 args=get(f,'UserData');
31 if isstruct(args)
32 UserDataArgsOK=isfield(args, 'SpacingHorizontal')& isfield(args,'Holdaxis')&isfield(args,'rows')&isfield(args,'cols');
33 end
34 OKToStoreArgs=isempty(args)|UserDataArgsOK;
35 if isempty(args)&&(~UserDataArgsOK)
36 args=struct('Holdaxis',0, ...
37 'SpacingVertical',0.05,'SpacingHorizontal',0.05, ...
38 'PaddingLeft',0,'PaddingRight',0,'PaddingTop',0,'PaddingBottom',0, ...
39 'MarginLeft',.1,'MarginRight',.1,'MarginTop',.1,'MarginBottom',.1, ...
40 'rows',[],'cols',[]);
41 end
42 args = parseArgs(varargin,args,{'Holdaxis'},{'Spacing' {'sh','sv'}; 'Padding' {'pl','pr','pt','pb'}; 'Margin' {'ml','mr','mt','mb'}});
43 if (length(args.NumericArguments)>2)
44 args.rows=args.NumericArguments{1};
45 args.cols=args.NumericArguments{2};
46 %remove these 2 numerical arguments
47 args.NumericArguments={args.NumericArguments{3:end}};
48 end
49 if OKToStoreArgs
50 set(f,'UserData',args);
51 end
52 switch length(args.NumericArguments)
53 case 0
54 return % no arguments but rows/cols....
55 case 1
56 if numel(args.NumericArguments{1}) > 1 % restore subplot(m,n,[x y]) behaviour
57 [x1 y1] = ind2sub([args.cols args.rows],args.NumericArguments{1}(1)); % subplot and ind2sub count differently (column instead of row first) --> switch cols/rows
58 [x2 y2] = ind2sub([args.cols args.rows],args.NumericArguments{1}(end));
59 else
60 x1=mod((args.NumericArguments{1}-1),args.cols)+1; x2=x1;
61 y1=floor((args.NumericArguments{1}-1)/args.cols)+1; y2=y1;
62 end
63 % x1=mod((args.NumericArguments{1}-1),args.cols)+1; x2=x1;
64 % y1=floor((args.NumericArguments{1}-1)/args.cols)+1; y2=y1;
65 case 2
66 x1=args.NumericArguments{1};x2=x1;
67 y1=args.NumericArguments{2};y2=y1;
68 case 4
69 x1=args.NumericArguments{1};x2=x1+args.NumericArguments{3}-1;
70 y1=args.NumericArguments{2};y2=y1+args.NumericArguments{4}-1;
71 otherwise
72 error('getAxes argument error')
73 end
74
75 cellwidth=((1-args.MarginLeft-args.MarginRight)-(args.cols-1)*args.SpacingHorizontal)/args.cols;
76 cellheight=((1-args.MarginTop-args.MarginBottom)-(args.rows-1)*args.SpacingVertical)/args.rows;
77 xpos1=args.MarginLeft+args.PaddingLeft+cellwidth*(x1-1)+args.SpacingHorizontal*(x1-1);
78 xpos2=args.MarginLeft-args.PaddingRight+cellwidth*x2+args.SpacingHorizontal*(x2-1);
79 ypos1=args.MarginTop+args.PaddingTop+cellheight*(y1-1)+args.SpacingVertical*(y1-1);
80 ypos2=args.MarginTop-args.PaddingBottom+cellheight*y2+args.SpacingVertical*(y2-1);
81 if args.Holdaxis
82 h=axes('position',[xpos1 1-ypos2 xpos2-xpos1 ypos2-ypos1]);
83 else
84 h=subplot('position',[xpos1 1-ypos2 xpos2-xpos1 ypos2-ypos1]);
85 end
86 set(h,'box','on');
87 %h=axes('position',[x1 1-y2 x2-x1 y2-y1]);
88 set(h,'units',get(gcf,'defaultaxesunits'));
89 set(h,'tag','getAxes');
90 if (nargout==0), clear h; end;
91end

93function argStruct = parseArgs(args, argStruct, varargin)
94 % Helper function for parsing varargin.
95 %
96 %
97 % argStruct=parseArgs(varargin,argStruct[,FlagtypeParams[,Aliases]])
98 %
99 % * argStruct is the structure full of named arguments with default values.
100 % * Flagtype params is params that don't require a value. (the value will be set to 1 if it is present)
101 % * Aliases can be used to map one argument-name to several argstruct fields
102 %
103 %
104 % example usage:
105 % --------------
106 % function parseargtest(varargin)
107 %
108 % %define the acceptable named arguments and assign default values
109 % args=struct('Holdaxis',0, ...
110 % 'SpacingVertical',0.05,'SpacingHorizontal',0.05, ...
111 % 'PaddingLeft',0,'PaddingRight',0,'PaddingTop',0,'PaddingBottom',0, ...
112 % 'MarginLeft',.1,'MarginRight',.1,'MarginTop',.1,'MarginBottom',.1, ...
113 % 'rows',[],'cols',[]);
114 %
115 % %The capital letters define abrreviations.
116 % % Eg. parseargtest('spacingvertical',0) is equivalent to parseargtest('sv',0)
117 %
118 % args=parseArgs(varargin,args, ... % fill the arg-struct with values entered by the user
119 % {'Holdaxis'}, ... %this argument has no value (flag-type)
120 % {'Spacing' {'sh','sv'}; 'Padding' {'pl','pr','pt','pb'}; 'Margin' {'ml','mr','mt','mb'}});
121 %
122 % disp(args)
123 %
124 %
125 %
126 %
127 % Aslak Grinsted 2004
128 % -------------------------------------------------------------------------
129 % Copyright (C) 2002-2004, Aslak Grinsted
130 % This software may be used, copied, or redistributed as long as it is not
131 % sold and this copyright notice is reproduced on each copy made. This
132 % routine is provided as is without any express or implied warranties
133 % whatsoever.
134 persistent matlabver
135 if isempty(matlabver)
136 matlabver=ver('MATLAB');
137 matlabver=str2double(matlabver.Version);
138 end
139 Aliases={};
140 FlagTypeParams='';
141 if (length(varargin)>0)
142 FlagTypeParams=lower(strvcat(varargin{1})); %#ok
143 if length(varargin)>1
144 Aliases=varargin{2};
145 end
146 end
147
148 %---------------Get "numeric" arguments
149 NumArgCount=1;
150 while (NumArgCount<=size(args,2))&&(~ischar(args{NumArgCount}))
151 NumArgCount=NumArgCount+1;
152 end
153 NumArgCount=NumArgCount-1;
154 if (NumArgCount>0)
155 argStruct.NumericArguments={args{1:NumArgCount}};
156 else
157 argStruct.NumericArguments={};
158 end
159 %--------------Make an accepted fieldname matrix (case insensitive)
160 Fnames=fieldnames(argStruct);
161 for i=1:length(Fnames)
162 name=lower(Fnames{i,1});
163 Fnames{i,2}=name; %col2=lower
164 Fnames{i,3}=[name(Fnames{i,1}~=name) ' ']; %col3=abreviation letters (those that are uppercase in the argStruct) e.g. SpacingHoriz->sh
165 %the space prevents strvcat from removing empty lines
166 Fnames{i,4}=isempty(strmatch(Fnames{i,2},FlagTypeParams)); %Does this parameter have a value?
167 end
168 FnamesFull=strvcat(Fnames{:,2}); %#ok
169 FnamesAbbr=strvcat(Fnames{:,3}); %#ok
170 if length(Aliases)>0
171 for i=1:length(Aliases)
172 name=lower(Aliases{i,1});
173 FieldIdx=strmatch(name,FnamesAbbr,'exact'); %try abbreviations (must be exact)
174 if isempty(FieldIdx)
175 FieldIdx=strmatch(name,FnamesFull); %&??????? exact or not?
176 end
177 Aliases{i,2}=FieldIdx;
178 Aliases{i,3}=[name(Aliases{i,1}~=name) ' ']; %the space prevents strvcat from removing empty lines
179 Aliases{i,1}=name; %dont need the name in uppercase anymore for aliases
180 end
181 %Append aliases to the end of FnamesFull and FnamesAbbr
182 FnamesFull=strvcat(FnamesFull,strvcat(Aliases{:,1})); %#ok
183 FnamesAbbr=strvcat(FnamesAbbr,strvcat(Aliases{:,3})); %#ok
184 end
185 %--------------get parameters--------------------
186 l=NumArgCount+1;
187 while (l<=length(args))
188 a=args{l};
189 if ischar(a)
190 paramHasValue=1; % assume that the parameter has is of type 'param',value
191 a=lower(a);
192 FieldIdx=strmatch(a,FnamesAbbr,'exact'); %try abbreviations (must be exact)
193 if isempty(FieldIdx)
194 FieldIdx=strmatch(a,FnamesFull);
195 end
196 if (length(FieldIdx)>1) %shortest fieldname should win
197 [mx,mxi]=max(sum(FnamesFull(FieldIdx,:)==' ',2));%#ok
198 FieldIdx=FieldIdx(mxi);
199 end
200 if FieldIdx>length(Fnames) %then it's an alias type.
201 FieldIdx=Aliases{FieldIdx-length(Fnames),2};
202 end
203
204 if isempty(FieldIdx)
205 error(['Unknown named parameter: ' a])
206 end
207 for curField=FieldIdx' %if it is an alias it could be more than one.
208 if (Fnames{curField,4})
209 if (l+1>length(args))
210 error(['Expected a value for parameter: ' Fnames{curField,1}])
211 end
212 val=args{l+1};
213 else %FLAG PARAMETER
214 if (l<length(args)) %there might be a explicitly specified value for the flag
215 val=args{l+1};
216 if isnumeric(val)
217 if (numel(val)==1)
218 val=logical(val);
219 else
220 error(['Invalid value for flag-parameter: ' Fnames{curField,1}])
221 end
222 else
223 val=true;
224 paramHasValue=0;
225 end
226 else
227 val=true;
228 paramHasValue=0;
229 end
230 end
231 if matlabver>=6
232 argStruct.(Fnames{curField,1})=val; %try the line below if you get an error here
233 else
234 argStruct=setfield(argStruct,Fnames{curField,1},val); %#ok <-works in old matlab versions
235 end
236 end
237 l=l+1+paramHasValue; %if a wildcard matches more than one
238 else
239 error(['Expected a named parameter: ' num2str(a)])
240 end
241 end
242end
