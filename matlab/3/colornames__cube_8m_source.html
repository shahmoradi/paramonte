1% Plot COLORNAMES palettes in a color cube (RGB/DIN99/Lab/LCh/HSV/XYZ). With DataCursor labels.
2%
3% (c) 2014-2019 Stephen Cobeldick
4%
5%%% Syntax:
6% colornames(palette,space)
7%
8% Plot the COLORNAMES palettes in an RGB/Lab/LCh/HSV/XYZ cube. Color names
9% can be viewed by clicking on the plotted points using the data cursor.
10%
11% Two vertical colorbars are displayed on the figure's right hand side,
12% showing the colormap in sequence, both in color and converted to grayscale.
13%
14% Note: Requires the function COLORNAMES and its associated MAT file (FEX 48155).
15%
16%% Input and Output Arguments %%
17%
18%%% Inputs (all inputs are optional):
19% palette = CharRowVector, the name of a palette supported by COLORNAMES.
20% space = CharRowVector, the colorspace to plot the palette in, e.g. 'RGB'.
21%
22%%% Outputs:
23% none
24%
25% See also COLORNAMES COLORNAMES_DELTAE COLORNAMES_VIEW MAXDISTCOLOR COLORMAP
26function colornames_cube(palette,space)
27
28%% Figure Parameters %%
29%
30persistent fgh axh lnh axc axg imc img txt rgb pah sph
31%
32%% Get COLORNAMES Palettes %%
33%
34[fnm,csf] = colornames();
35%
36if nargin<1
37 idp = 1+rem(round(now*1e7),numel(fnm));
38else
39 assert(ischar(palette)&&isrow(palette),'First input <palette> must be a 1xN char.')
40 idz = strcmpi(palette,fnm);
41 assert(any(idz),'Palette ''%s'' is not supported. Call COLORNAMES() to list all palettes.',palette)
42 idp = find(idz);
43end
44%
45%% Color Space List %%
46%
47clm = {... % axes limits
48 {[0,1],[0,1],[0,1]},... RGB
49 {[0,100],[-Inf,+Inf],[-Inf,+Inf]},... DIN99
50 {[0,100],[-Inf,+Inf],[-Inf,+Inf]},... Lab
51 {[0,100],[0,+Inf],[0,360]},... LCh
52 {[0,360],[0,1],[0,1]},... HSV
53 {[0,1],[0,1],[0,1]}}; % XYZ
54lbl = {... % axes labels
55 {'Red','Green','Blue'},... RGB
56 {'L_{99}','a_{99}','b_{99}'},... DIN99
57 {'L*','a*','b*'},... Lab
58 {'Lightness','Chroma','Hue'},... LCh
59 {'Hue','Saturation','Value'},... HSV
60 {'X','Y','Z'}}; % XYZ
61z90 = [ true, false, false, true, true, false]; % 90 degree Z label.
62aar = [ false, false, false, true, true, false]; % automatic axes ratio.
63csp = { 'RGB','DIN99', 'Lab', 'LCh', 'HSV', 'XYZ'}; % colorspace.
64xyz = {[3,2,1],[3,2,1],[3,2,1],[3,2,1],[1,2,3],[1,2,3]}; % axis order.
65%
66if nargin<2
67 ids = 1+rem(round(now*1e7),numel(csp));
68else
69 assert(ischar(space)&&isrow(space),'Second input <sort> must be a 1xN char.')
70 tmp = strcmpi(space,csp);
71 assert(any(tmp),'Second input must be one of:%s\b.',sprintf(' %s,',csp{:}))
72 ids = find(tmp);
73end
74%
75%% Create a New Figure %%
76%
77if isempty(fgh) || ~ishghandle(fgh)
78 % Create figure and main axes:
79 fgh = figure('HandleVisibility','callback', 'IntegerHandle','off',...
80 'NumberTitle','off', 'Name',mfilename, 'Color','white', 'Toolbar','figure');
81 axh = axes('Parent',fgh, 'NextPlot','replacechildren', 'View',[55,32]);
82 grid(axh,'on')
83 % Create colorbars:
84 axc = axes('Parent',fgh, 'Units','normalized', 'Position',[0.96,0,0.02,1],...
85 'Visible','off', 'YLim',[0,1], 'HitTest','off');
86 axg = axes('Parent',fgh, 'Units','normalized', 'Position',[0.98,0,0.02,1],...
87 'Visible','off', 'YLim',[0,1], 'HitTest','off');
88 imc = image('CData',[0.25;0.5;0.75], 'Parent',axc);
89 img = image('CData',[0.75;0.5;0.25], 'Parent',axg);
90 txt = uicontrol(fgh, 'Units','Pixels', 'Position',[0,0,30,15], 'Style','text');
91 uicontrol(fgh, 'Units','Normalized', 'Position',[0.88,0.96,0.08,0.04],...
92 'Style','togglebutton', 'Callback',@cncDemoClBk, 'String','Demo');
93 % Create space and palette menus:
94 pah = uicontrol(fgh, 'Units','normalized', 'Position',[0,0.95,0.15,0.05],...
95 'Style','popupmenu', 'Callback',@cncScmClBk, 'String',fnm);
96 sph = uicontrol(fgh, 'Units','normalized', 'Position',[0,0.90,0.10,0.05],...
97 'Style','popupmenu', 'Callback',@cncSpcClBk, 'String',csp);
98 % Add DataCursor labels:
99 dcm = datacursormode(fgh);
100 set(dcm,'UpdateFcn',@(o,e)get(get(e,'Target'),'UserData'));
101 datacursormode(fgh,'on')
102end
103set(pah,'Value',idp);
104set(sph,'Value',ids);
105%
106%% Callback Functions %%
107%
108 function cncScmClBk(h,~) % Palette Callback
109 idp = get(h,'Value');
110 cncMapPlot()
111 cncClrSpace()
112 end
113%
114 function cncSpcClBk(h,~) % Color Space Callback
115 ids = get(h,'Value');
116 cncClrSpace()
117 end
118%
119%% Re/Draw Points in 3D Plot %%
120%
121 function cncMapPlot()
122 % Delete any existing colors:
123 delete(lnh(ishghandle(lnh)))
124 % Get new colors:
125 [cnc,rgb] = colornames(fnm{idp});
126 N = numel(cnc);
127 mag = rgb*[0.298936;0.587043;0.114021];
128 % Update main axes:
129 set(axh, 'ColorOrder',rgb, 'NextPlot','replacechildren');
130 % Update colorbars:
131 set(axc, 'YLim',[0,N]+0.5)
132 set(axg, 'YLim',[0,N]+0.5)
133 set(imc, 'CData',permute(rgb,[1,3,2]))
134 set(img, 'CData',repmat(mag,[1,1,3]))
135 set(txt, 'String',num2str(N))
136 % Plot each node:
137 idx = xyz{ids};
138 map = rgb;
139 map(:,:,2) = NaN;
140 map = permute(map,[3,1,2]);
141 lnh = plot3(map(:,:,idx(1)),map(:,:,idx(2)),map(:,:,idx(3)),...
142 '.','MarkerSize',36, 'Parent',axh);
143 % Add DataCursor labels:
144 arrayfun(@(h,n)set(h,'UserData',n{1}), lnh, cnc(:));
145 end
146%
147 function cncClrSpace()
148 % Plot the data in the requested color space.
149 switch csp{ids}
150 case 'RGB'
151 map = rgb;
152 case 'HSV'
153 map = csf.rgb2hsv(rgb);
154 case 'XYZ'
155 map = csf.rgb2xyz(rgb);
156 case 'Lab'
157 map = csf.xyz2lab(csf.rgb2xyz(rgb));
158 case 'LCh'
159 map = csf.lab2lch(csf.xyz2lab(csf.rgb2xyz(rgb)));
160 case 'DIN99'
161 map = csf.lab2d99(csf.xyz2lab(csf.rgb2xyz(rgb)));
162 otherwise
163 error('Sorry, the colorspace "%s" is not recognized.',csp{ids})
164 end
165 %
166 idx = xyz{ids};
167 lab = lbl{ids};
168 lim = clm{ids};
169 %
170 set(lnh,{'XData','YData','ZData'},num2cell(map(:,idx)))
171 %
172 xlabel(axh,lab{idx(1)})
173 ylabel(axh,lab{idx(2)})
174 zlabel(axh,lab{idx(3)}, 'Rotation',90*z90(ids))
175 xlim(axh,lim{idx(1)})
176 ylim(axh,lim{idx(2)})
177 zlim(axh,lim{idx(3)})
178 %
179 if aar(ids)
180 set(axh,'DataAspectRatioMode','auto')
181 else
182 set(axh,'DataAspectRatio',[1,1,1])
183 end
184 %
185 drawnow()
186 end

188%% Demonstration Function %%
189%
190 function cncDemoClBk(h,~)
191 % Slowly rotate the axes while the toggle button is depressed.
192 stp = 2;
193 itr = 0;
194 while ishghandle(h) && get(h,'Value')
195 if itr<=0
196 itr = randi([45,360])/stp;
197 ang = 360*rand(1);
198 dth = stp*sind(ang);
199 dph = stp*cosd(ang);
200 end
201 cncOrbit(axh,dth,dph)
202 itr = itr-1;
203 % Wait a smidgen:
204 pause(0.07)
205 end
206 end
207%
208%% Initialize the Figure %%
209%
210rgb = [];
211cnc = {};
212cncMapPlot()
213cncClrSpace()
214%
215end

217function cncOrbit(axh,dth,dph)
218% Rotate camera around the center of the axes. Avoids bug in CAMROTATE.
219%
220C = get(axh, {'CameraPosition','CameraTarget','CameraUpVector','DataAspectRatio'});
221%
222vec = (C{2}-C{1})./C{4};
223vax = cross(vec, C{3}./C{4});
224hax = cross(vax, vec);
225%
226vec = vec/norm(vec);
227vax = vax/norm(vax);
228hax = hax/norm(hax);
229%
230cosa = cosd(dth);
231sina = sind(dth);
232vera = 1 - cosa;
233rotH = [...
234 cosa+hax(1)^2*vera,hax(1)*hax(2)*vera-hax(3)*sina,hax(1)*hax(3)*vera+hax(2)*sina;...
235 hax(1)*hax(2)*vera+hax(3)*sina,cosa+hax(2)^2*vera,hax(2)*hax(3)*vera-hax(1)*sina;...
236 hax(1)*hax(3)*vera-hax(2)*sina,hax(2)*hax(3)*vera+hax(1)*sina,cosa+hax(3)^2*vera]';
237%
238cosa = cosd(-dph);
239sina = sind(-dph);
240vera = 1 - cosa;
241rotV = [...
242 cosa+vax(1)^2*vera,vax(1)*vax(2)*vera-vax(3)*sina,vax(1)*vax(3)*vera+vax(2)*sina;...
243 vax(1)*vax(2)*vera+vax(3)*sina,cosa+vax(2)^2*vera,vax(2)*vax(3)*vera-vax(1)*sina;...
244 vax(1)*vax(3)*vera-vax(2)*sina,vax(2)*vax(3)*vera+vax(1)*sina,cosa+vax(3)^2*vera]';
245%
246rotM = rotV*rotH;
247%
248pos = (-vec*rotM).*C{4} + C{2};
249upv = (+hax*rotM).*C{4};
250%
251set(axh,'CameraPosition',pos, 'CameraUpVector',upv)
252%
253end
