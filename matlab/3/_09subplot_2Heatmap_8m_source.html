47 %> \brief
48 %> Reset the heatmap colormap limits to the user-specified limits ``lb, ub``.
49 %> If either is empty or missing, keep the existing limit for the corresponding empty limit.
50 %> If both are empty or missing, symmetrize the existing limits.<br>
51 %> The specified input values will be used to set the ``ColorLimits``
52 %> component of the Heatmap object as ``ColorLimits = [lb, ub]``.<br>
53 %> If both input values are missing, the current colormap
54 %> range of the Heatmap plots will be symmetrized.
55 %>
56 %> \warning
57 %> This method has side-effects by manipulating
58 %> the existing attributes of the parent object.
59 %>
60 %> \param[in] lb : The input MATLAB object that can be either:
61 %> <ol>
62 %> <li> a scalar of type double representing the lower bound of the Heatmap colormap range.
63 %> <li> a vector of type double of size ``2`` representing the
64 %> lower and upper bounds of the Heatmap colormap range.
65 %> </ol>
66 %> (**optional**. If missing, the current value will remain intact.)

104 function setColorLim(self, lb, ub)
105
106 if isfield(self.fout, "heatmap") && isprop(self.fout.heatmap, "ColorLimits")
107 if nargin < 3
108 ub = [];
109 end
110 if nargin < 2
111 lb = [];
112 elseif 1 < length(lb)
113 ub = lb(2);
114 lb = lb(1);
115 end
116 if isempty(lb) && isempty(ub)
117 % symmetrize the existing range.
118 maxval = max(abs(self.fout.heatmap.ColorLimits));
119 limits = [-maxval, maxval];
120 else
121 limits = self.fout.heatmap.ColorLimits;
122 if ~isempty(lb)
123 limits(1) = lb;
124 end
125 if ~isempty(ub)
126 limits(2) = ub;
127 end
128 end
129 self.fout.heatmap.ColorLimits = limits;
130 else
131 error ( newline ...
132 + "There is no component ``fout.heatmap.ColorLimits`` for this object to adjust the limits." + newline ...

135 end
