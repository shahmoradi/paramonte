34function itis = isgui()
35 persistent itis_persistent
36 if isempty(itis_persistent)
37 try
38 % usejava() is sensitive to char vs. string input arguments. always input char.
39 itis_persistent = usejava('desktop') && feature('ShowFigureWindows') && ~batchStartupOptionUsed; % batchStartupOptionUsed is introduced in R2019a and not supported in older versions of MATLAB
40 catch
41 itis_persistent = usejava('desktop') && feature('ShowFigureWindows');
42 end
43 end
44 itis = itis_persistent;
