1% Convert between RGB values and color names: from RGB to names, and names to RGB.
2%
3% (c) 2014-2019 Stephen Cobeldick
4%
5% Easily convert between RGB values and color names, in both directions!
6%
7% COLORNAMES matches the input colors (either names or an RGB map) to colors
8% from the requested palette. It returns both the color names and RGB values.
9%
10%%% Syntax:
11% palettes = colornames()
12% names = colornames(palette)
13% names = colornames(palette,RGB)
14% names = colornames(palette,RGB,deltaE)
15% names = colornames(palette,names)
16% names = colornames(palette,name1,name2,...)
17% [names,RGB] = colornames(palette,...)
18%
19%%% RGB inputs:
20% * Accepts multiple RGB values in a standard MATLAB Nx3 colormap.
21% * Choice of color distance (deltaE) calculation, any one of:
22% 'CIEDE2000', 'DIN99', 'CIE94' (default), 'CIE76', 'CMCl:c', or 'RGB'.
23% For info on deltaE: <https://en.wikipedia.org/wiki/Color_difference>
24% or: <http://www.colorwiki.com/wiki/Delta_E:_The_Color_Difference>
25% Note that palettes with sparse colors can produce unexpected matches.
26%
27%%% Color name inputs:
28% * Accepts multiple color names (in one cell array, or as separate inputs).
29% * Case-insensitive color name matches, e.g. 'Blue' == 'blue' == 'BLUE'.
30% * Allows optional space characters between the color name words.
31% * Allows camelCase to specify color names with space characters in them.
32% * Palettes with index numbers may be specified by the number: e.g.: '5'.
33% * Palettes Alphabet, MATLAB, and Natural also match the initial letter to
34% the color name (except for 'Black' which is matched by 'k').
35%
36%% Space Characters in Color Names %%
37%
38% Many palettes use camelCase in the color names: COLORNAMES will match
39% the input names with any character case or spaces between the words, e.g.:
40% 'Sky Blue' == 'SKY BLUE' == 'sky blue' == 'SkyBlue' == 'SKYBLUE' == 'skyblue'.
41%
42% Palettes Foster and xkcd include spaces: clashes occur if the names are
43% converted to one case (e.g. lower) and the spaces removed. To make these
44% names more convenient to use, camelCase is equivalent to words separated
45% by spaces, e.g.: 'EggShell' == 'Egg Shell' == 'egg shell' == 'EGG SHELL'.
46% Note this is a different color to 'Eggshell' == 'eggshell' == 'EGGSHELL'.
47%
48% In xkcd the forward slash ('/') also distinguishes between different
49% colors, e.g.: 'Blue/Green' is not the same as 'Blue Green' (== 'BlueGreen').
50%
51%% Index Numbers in Color Names %%
52%
53% Palettes with a leading index number (e.g. AppleII, BS381C, CGA, RAL, etc)
54% can also use just the index number or words to select a color, e.g.:
55% '5' == 'Blue Flower' == 'BLUE FLOWER' == 'BlueFlower' == '5 Blue Flower'
56% And for the palettes with spaces, camelCase is also equivalent to words
57% separated by spaces, e.g.: '5BlueFlower' == '5 Blue Flower' == '5 blue flower'
58%
59%% Initial Letter Color Name Abbreviations %%
60%
61% Palettes Alphabet, MATLAB, and Natural also match the initial letter to
62% the color name (except for 'Black' which is matched by 'k'), e.g.:
63% 'B' == 'Blue', 'Y' =='Yellow', 'M' == 'Magenta', 'K' == 'Black'.
64%
65%% Examples %%
66%
67% >> palettes = colornames()
68% palettes =
69% 'Alphabet'
70% 'AmstradCPC'
71% 'AppleII'
72% 'Bang'
73% 'BS381C'
74% 'CGA'
75% 'Crayola'
76% 'CSS'
77% 'dvips'
78% 'Foster'
79% 'HTML4'
80% 'ISCC'
81% 'Kelly'
82% 'MacBeth'
83% 'MATLAB'
84% 'Natural'
85% 'R'
86% 'RAL'
87% 'Resene'
88% 'Resistor'
89% 'SherwinWilliams'
90% 'SVG'
91% 'Tableau'
92% 'Thesaurus'
93% 'Trubetskoy'
94% 'Wikipedia'
95% 'Wolfram'
96% 'X11'
97% 'xcolor'
98% 'xkcd'
99%
100% >> colornames('Natural') % all color names for one palette
101% ans =
102% 'Black'
103% 'Blue'
104% 'Green'
105% 'Red'
106% 'White'
107% 'Yellow'
108%
109% >> [names,rgb] = colornames('HTML4','blue','red','teal','olive')
110% names =
111% 'Blue'
112% 'Red'
113% 'Teal'
114% 'Olive'
115% rgb =
116% 0 0 1.0000
117% 1.0000 0 0
118% 0 0.5020 0.5020
119% 0.5020 0.5020 0
120%

122% ans =
123% 'Blue'
124% 'Red'
125%
126% >> colornames('HTML4',[0,0.5,1;1,0.5,0],'rgb') % specify deltaE
127% ans =
128% 'Teal'

136% Cyan 0 1 1
137% Green 0 1 0
138% Magenta 1 0 1
139% Red 1 0 0
140% White 1 1 1

150%% Input and Output Arguments %%
151%
152%%% Inputs (*=default):
153% palette = CharRowVector, the name of a supported palette, e.g.: 'CSS'.
154%%% The optional input/s can be names or RGB values. Names can be either:
155% names = CellOfCharRowVectors, any number of supported color names.
156% name1,name2,... = CharRowVectors, any number of supported color names.
157%%% RGB values in a matrix, with optional choice of color-distance deltaE:
158% RGB = NumericMatrix, size Nx3, each row is an RGB triple (0<=rgb<=1).
159% deltaE = CharRowVector: 'CIEDE2000', 'DIN99', 'CIE94'*, 'CIE76', 'CMCl:c', or 'RGB'.
160%
161%%% Outputs:
162% cnc = CellOfCharRowVectors, size Nx1, the color names that best match the inputs.
163% rgb = NumericMatrix, size Nx3, RGB values corresponding to names in <cnc>.
164%
165% See also COLORNAMES_CUBE COLORNAMES_DELTAE COLORNAMES_VIEW MAXDISTCOLOR COLORMAP
166function [cnc, rgb, dtE] = colornames(palette, varargin)
167
168 %% Read Palette Data %%
169 %
170 persistent data
171 %
172 if isempty(data)
173 data = load('colornames.mat');
174 end
175 %
176 fnc = fieldnames(data);
177 dtE = {'CIEDE2000','DIN99','CIE94','CIE76','CMCl:c','RGB'};
178 %
179 %% Return All Palette Names %%
180 %
181 if nargin==0
182 if nargout
183 cnc = fnc;
184 rgb = struct('invgamma',@cnGammaInv, 'lab2d99',@cnLab2d99,...
185 'rgb2hsv',@cnRGB2HSV, 'rgb2xyz',@cnRGB2XYZ,...
186 'xyz2lab',@cnXYZ2Lab, 'lab2lch',@cnLab2LCh);
187 else
188 out = reshape(fnc,1,[]);
189 out(2,:) = cellfun(@(p)numel(data.(p).names),fnc,'Uni',0);
190 out(3,:) = cellfun(@(p)data.(p).notes,fnc,'Uni',0);
191 out(5,:) = cellfun(@(p)data.(p).source,fnc,'Uni',0);
192 out(4,:) = {repmat(' ',1,32)};
193 fprintf('%23s %5d %-s\n%sSource: %-s\n',out{:}) %#ok<PRTCAL>
194 end
195 return
196 end
197 %
198 %% Retrieve a Palette's Color Names and RGB %%
199 %
200 ord = {'first','second','third'};
201 %
202 assert(ischar(palette)&&isrow(palette),'The %s input must be a 1xN char.',ord{1})
203 idp = strcmpi(palette,fnc);
204 assert(any(idp),'Palette ''%s'' is not supported. Call COLORNAMES() to list all palettes.',palette)
205 pss = data.(fnc{idp}); % the requested palette.
206 %
207 cnc = pss.names;
208 rgb = double(pss.rgb) ./ pss.scale;
209 %
210 %% Match Input RGB to Palette Colors %%
211 %
212 isCoS = @(C)cellfun('isclass',C,'char') & cellfun('size',C,1)<2 & cellfun('ndims',C)<3;
213 %
214 if nargin==1 % return complete palette
215 return
216 elseif isnumeric(varargin{1}) % RGB values
217 idx = cnClosest(dtE,ord,rgb,varargin{1},varargin(2:end));
218 cnc = cnc(idx,:);
219 rgb = rgb(idx,:);
220 return
221 elseif iscell(varargin{1}) % color names in a cell array
222 assert(nargin==2,'Too many inputs: only one cell array of color names is allowed.')
223 inp = varargin{1}(:);
224 assert(all(isCoS(inp)),'Every cell element must contain one char row vector (1xN char).')
225 else % individual color names
226 inp = varargin(:);
227 assert(all(isCoS(inp)),'Input arguments must be char row vectors (1xN char), or an RGB map (Nx3 num).')
228 end
229 %
230 %% Match Input Names to Palette Color Names %%
231 %
232 tkn = regexp(pss.names,'^([-+]?\d+)?\s*(.+)$','once','tokens');
233 if all(cellfun('size',tkn,1)>1) % Octave
234 tkn = horzcat(tkn{:}).';
235 else % MATLAB & Octave
236 tkn = vertcat(tkn{:});
237 end
238 % Leading indices (Octave || MATLAB):
239 if size(tkn,2)<2 || any(cellfun('isempty',tkn(:,1)))
240 inx = {};
241 else
242 inx = tkn(:);
243 end
244 % Initial letters:
245 inq = upper(regexp(tkn(:,end),'^[A-Za-z]','once','match'));
246 inq(strcmpi(tkn(:,end),'Black')) = {'K'};
247 if numel(pss.names)~=numel(unique(inq))
248 inq = {};
249 end
250 %
251 % Convert palette color names to regular expressions:
252 rgx = strcat('^',regexptranslate('escape',[cnc;inx;inq]),'$');
253 % Split camelCase color names into separate words with optional space character:
254 rgx = regexprep(rgx,{'([a-z])([A-Z])',' ','([Gg])r[ae]y'},{'$1 $2',' ?','$1r[ae]y'});
255 % Use the regular expressions to match input names to palette color names:
256 idm = cellfun(@(s)~cellfun('isempty',regexpi(s,rgx,'once')),inp,'UniformOutput',false);
257 idn = cellfun(@nnz,idm);
258 % Any unmatched input names throw an error:
259 if any(~idn)
260 cnNoMatch(fnc{idp},cnc,inp(idn==0))
261 end
262 % If any input name matches multiple palette regexps, pick the best match:
263 idz = idn>1;
264 fun = @(s,m)cnPickBest([cnc;inx;inq],s,m);
265 idm(idz) = cellfun(fun, inp(idz),idm(idz), 'UniformOutput',false);
266 % Get palette indices of matched input names:
267 idx = cellfun(@find,idm);
268 idx = 1+mod(idx-1,numel(cnc)); % accounts for initials & leading indices.
269 %
270 cnc = cnc(idx,:);
271 rgb = rgb(idx,:);
272 %
273end

277function idm = cnPickBest(raw, str, idm)
278 % Pick the closest color name to match the given input name <str>.
279 %
280 str = regexprep(str,'([a-z])([A-Z])','$1 $2');
281 nme = cellfun(@(s)cnEdits(s,str),raw(idm));
282 [~,ide] = min(nme); % MIN guarantees exactly one index is returned.
283 idf = find(idm);
284 idm(:) = false;
285 idm(idf(ide)) = true;
286end

290function cnNoMatch(pnm,cnc,inp)
291 % Find palette color names closest to <inp> names, and print an error message.
292 % Sort names into order of closest to furthest match, measure name lengths:
293 [~,idx] = cellfun(@(s)sort(cellfun(@(t)cnEdits(s,t),cnc)),inp,'UniformOutput',false);
294 lft = cellfun(@(s)sprintf('%-23s ->',s),inp,'UniformOutput',false);
295 cus = cellfun(@(s,x)cumsum([2+numel(s);4+cellfun('length',cnc(x))]),lft,idx,'UniformOutput',false);
296 try
297 cws = get(0,'CommandWindowSize');
298 catch %#ok<CTCH>
299 cws = matlab.desktop.commandwindow.size;
300 end
301 eos = @(s)[s(1:end-1),'.'];
302 % Join names together, taking into account the command window width:
303 idn = cellfun(@(v)find(v<=max(99,cws(1)),1,'last'),cus,'UniformOutput',false);
304 rgt = cellfun(@(x,n)sprintf(' ''%s'',',cnc{x(1:n-1)}),idx,idn,'UniformOutput',false);
305 tmp = cellfun(eos, strcat(lft,rgt), 'UniformOutput',false);
306 %
307 str = sprintf('%s\n',tmp{:});
308 str = sprintf('Palette color names that are similar to the input names:\n%s',str);
309 %
310 % Print error message with color names similar to the input names:
311 pnm = sprintf('''%s''',pnm);
312 txt = eos(sprintf(' ''%s'',',inp{:}));
313 error(['The palette %1$s does not contain these colors:%2$s\n\n%3$s\n'...
314 'Call COLORNAMES(%1$s) to list all color names for that palette,\n'...
315 'or COLORNAMES_VIEW(%1$s) to view the palette in a 2D list,\n'...
316 'or COLORNAMES_CUBE(%1$s) to view the palette in a 3D cube.\n'],pnm,txt,str) %#ok<SPERR>
317end

321function d = cnEdits(S1,S2)
322 % Wagner-Fischer algorithm to calculate the Levenshtein edit distance.
323 N1 = 1+numel(S1);
324 N2 = 1+numel(S2);
325 D = zeros(N1,N2);
326 D(:,1) = 0:N1-1;
327 D(1,:) = 0:N2-1;
328 for r = 2:N1
329 for c = 2:N2
330 D(r,c) = min([D(r-1,c)+1, D(r,c-1)+1, D(r-1,c-1)+~strcmpi(S1(r-1),S2(c-1))]);
331 end
332 end
333 d = D(end);
334end

338function idx = cnClosest(dtE,ord,rgb,inp,typ)
339 % Use color difference (deltaE) to identify the closest colors to input RGB values.
340 dtS = sprintf(' ''%s'',',dtE{:});
341 switch numel(typ)
342 case 0
343 typ = 'CIE94';
344 case 1
345 typ = typ{1};
346 assert(ischar(typ)&&isrow(typ),[...
347 'If the %s argument is an RGB map, then the optional %s\n'...
348 'argument can select the color difference (deltaE) metric.\n'...
349 'It must be one of:%s\b'],ord{2},ord{3},dtS)
350 otherwise
351 error('Too many input arguments. See help for information on input options.')
352 end
353 assertion = ismatrix(inp) && size(inp, 2) == 3 && isreal(inp) && all(0 <= inp(:) & inp(:) <= 1);
354 assert(assertion, 'The %s input argument can be a map of RGB values (size Nx3).', ord{2});
355 % Calculate the Color Difference (deltaE) %%
356 if strcmpi(typ,'RGB')
357 [~,idx] = cellfun(@(v)min(sum(bsxfun(@minus,rgb,v).^2,2)),num2cell(inp,2));
358 return
359 end
360 lab = cnXYZ2Lab(cnRGB2XYZ(rgb));
361 inp = cnXYZ2Lab(cnRGB2XYZ(inp));
362 switch upper(typ)
363 case 'CIEDE2000'
364 [~,idx] = cellfun(@(v)min(sum(cnCIE2k(lab,v),2)),num2cell(inp,2));
365 case 'CIE94'
366 [~,idx] = cellfun(@(v)min(sum(cnCIE94(lab,v),2)),num2cell(inp,2));
367 case 'CMCL:C'
368 [~,idx] = cellfun(@(v)min(sum(cnCMClc(lab,v),2)),num2cell(inp,2));
369 case {'CIE76','LAB'}
370 [~,idx] = cellfun(@(v)min(sum(bsxfun(@minus,lab,v).^2,2)),num2cell(inp,2));
371 case 'DIN99'
372 [~,idx] = cellfun(@(v)min(sum(bsxfun(@minus,cnLab2d99(lab),v).^2,2)),num2cell(cnLab2d99(inp),2));
373 otherwise
374 error(['The %s input, color difference (deltaE) metric ''%s'', is not supported.'...
375 '\nThe supported color difference metrics are:%s\b'],ord{3},typ,dtS)
376 end
377end

381function rgb = cnGammaInv(rgb)
382 % Inverse gamma transform of sRGB data, Nx3 RGB -> Nx3 RGB.
383 idx = rgb <= 0.04045;
384 rgb(idx) = rgb(idx) / 12.92;
385 rgb(~idx) = real(((rgb(~idx) + 0.055) / 1.055).^2.4);
386end

402function lab = cnXYZ2Lab(xyz)
403 % Convert an Nx3 matrix of XYZ values to an Nx3 matrix of CIELab values.
404 xyz = bsxfun(@rdivide,xyz,[0.95047,1,1.08883]); % D65
405 idx = xyz>(6/29)^3;
406 F = idx.*(xyz.^(1/3)) + ~idx.*(xyz*(29/6)^2/3+4/29);
407 % Remember to include my license when copying my implementation.
408 lab(:,2:3) = bsxfun(@times,[500,200],F(:,1:2)-F(:,2:3));
409 lab(:,1) = 116*F(:,2) - 16;
410end

414function Lab99 = cnLab2d99(Lab)
415 % Convert an Nx3 matrix of CIELab values to Nx3 matrix of DIN99 values.
416 L99 = 105.51 * log(1 + 0.0158*Lab(:,1));
417 e = Lab(:,2).*cosd(16)+Lab(:,3).*sind(16);
418 f = 0.7*(Lab(:,3).*cosd(16)+Lab(:,2).*sind(16));
419 G = sqrt(e.^2 + f.^2);
420 C99 = log(1 + 0.045*G)./0.045;
421 h99 = atan2(f,e);
422 a99 = C99 .* cos(h99);
423 b99 = C99 .* sin(h99);
424 Lab99 = [L99,a99,b99];
425 % Remember to include my license when copying my implementation.
426end

440function hsv = cnRGB2HSV(rgb)
441 % Convert an Nx3 matrix of sRGB triples to an Nx3 matrix of HSV values.
442 rgb = cnGammaInv(rgb);
443 [V,X] = max(rgb,[],2);
444 S = V - min(rgb,[],2);
445 N = numel(S);
446 L = N*mod(X+0,3) + (1:N).';
447 R = N*mod(X+1,3) + (1:N).';
448 % Remember to include my license when copying my implementation.
449 H = mod(2*(X-1)+(rgb(L)-rgb(R))./S,6);
450 S = S./V;
451 S(V==0) = 0;
452 H(S==0) = 0;
453 hsv = [60*H,S,V];
454end

458function LCHs = cnCIE94(mat,one)
459 % Convert Nx3 and 1x3 Lab values into Nx3 CIE94 deltaE^2 values.
460 kLCH = [2,1,1]; % [1,1,1]
461 K012 = [0,0.048,0.014]; % [0,0.045,0.015]
462 Ca1 = sqrt(sum(mat(:,[2,3]).^2,2));
463 Ca2 = sqrt(sum(one(:,[2,3]).^2,2));
464 % Remember to include my license when copying my implementation.
465 dHa = sqrt((mat(:,2)-one(:,2)).^2 + (mat(:,3)-one(:,3)).^2 - (Ca1-Ca2).^2);
466 LCHs = ([(mat(:,1)-one(:,1)), (Ca1-Ca2), dHa] ./ bsxfun(@times, kLCH, (1 + bsxfun(@times, Ca1, K012)))).^2;
467end

471function LCHsR = cnCIE2k(mat,one)
472 % Convert Nx3 and 1x3 Lab values into Nx4 CIEDE2000 deltaE^2 values.
473 kLCH = [1,1,1];
474 Ca1 = sqrt(sum(mat(:,2:3).^2,2));
475 Ca2 = sqrt(sum(one(:,2:3).^2,2));
476 % Remember to include my license when copying my implementation.
477 Cb = (Ca1+Ca2)/2;
478 Lb = (mat(:,1)+one(:,1))/2;
479 tmp = 1-sqrt(Cb.^7 ./ (Cb.^7 + 25^7));
480 ap1 = mat(:,2) .* (1+tmp/2);
481 ap2 = one(:,2) .* (1+tmp/2);
482 Cp1 = sqrt(ap1.^2 + mat(:,3).^2);
483 Cp2 = sqrt(ap2.^2 + one(:,3).^2);
484 Cbp = (Cp1+Cp2)/2;
485 Cpp = Cp1.*Cp2;
486 idx = Cpp==0;
487 hp1 = cnAtan2d(mat(:,3),ap1);
488 hp2 = cnAtan2d(one(:,3),ap2);
489 dhp = 180-mod(180+hp1-hp2,360);
490 dhp(idx) = 0;
491 Hbp = mod((hp1+hp2)/2 - 180*(abs(hp1-hp2)>180),360);
492 Hbp(idx) = hp1(idx)+hp2(idx);
493 T = 1-0.17*cosd(Hbp-30)+0.24*cosd(2*Hbp)+0.32*cosd(3*Hbp+6)-0.2*cosd(4*Hbp-63);
494 RT = -sind(60*exp(-((Hbp-275)/25).^2)) .* sqrt(Cbp.^7 ./ (Cbp.^7 + 25^7))*2;
495 SLCH = [(0.015*(Lb-50).^2)./sqrt(20+(Lb-50).^2), 0.045*Cbp, 0.015*Cbp.*T];
496 dLCH = [(one(:,1)-mat(:,1)), (Cp2-Cp1), 2*sqrt(Cpp).*sind(dhp/2)] ./ bsxfun(@times, kLCH, 1 + SLCH);
497 LCHsR = [dLCH.^2, RT.*prod(dLCH(:,2:3),2)];
498end

502function LCHs = cnCMClc(mat,one)
503 % Convert Nx3 and 1x3 Lab values into Nx3 CMC l:c deltaE^2 values.
504 Ca1 = sqrt(sum(mat(:,[2,3]).^2,2));
505 Ca2 = sqrt(sum(one(:,[2,3]).^2,2));
506 % Remember to include my license when copying my implementation.
507 dHa = sqrt((mat(:,2)-one(:,2)).^2 + (mat(:,3)-one(:,3)).^2 - (Ca1-Ca2).^2);
508 SL = 0.040975*mat(:,1) ./ (1+0.01765*mat(:,1));
509 SL(mat(:,1)<16) = 0.511;
510 SC = 0.0638 * (1 + Ca1./(1+0.0131*Ca1));
511 h1 = cnAtan2d(mat(:,3),mat(:,2));
512 F = sqrt(Ca1.^4./(1900+Ca1.^4));
513 A = [0.36;0.56]; B = [0.4;0.2]; D = [35;168];
514 X = 1+(164<=h1 & h1<=345);
515 T = A(X) + abs(B(X).*cosd(h1+D(X)));
516 SH = SC .* (F.*T + 1 - F);
517 LCHs = ([(one(:,1)-mat(:,1)), (Ca2-Ca1), dHa] ./ [SL,SC,SH]).^2;
518end
