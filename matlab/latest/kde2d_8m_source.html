2function [bandwidth, density, X, Y] = kde2d(data, n, MIN_XY, MAX_XY)
3 % fast and accurate state-of-the-art
4 % bivariate kernel density estimator
5 % with diagonal bandwidth matrix.
6 % The kernel is assumed to be Gaussian.
7 % The two bandwidth parameters are
8 % chosen optimally without ever
9 % using/assuming a parametric model for the data or any "rules of thumb".
10 % Unlike many other procedures, this one
11 % is immune to accuracy failures in the estimation of
12 % multimodal densities with widely separated modes (see examples).
13 % INPUTS: data - an N by 2 array with continuous data
14 % n - size of the n by n grid over which the density is computed
15 % n has to be a power of 2, otherwise n=2^ceil(log2(n));
16 % the default value is 2^8;
17 % MIN_XY,MAX_XY- limits of the bounding box over which the density is computed;
18 % the format is:
19 % MIN_XY=[lower_Xlim,lower_Ylim]
20 % MAX_XY=[upper_Xlim,upper_Ylim].
21 % The dafault limits are computed as:
22 % MAX=max(data,[],1); MIN=min(data,[],1); Range=MAX-MIN;
23 % MAX_XY=MAX+Range/4; MIN_XY=MIN-Range/4;
24 % OUTPUT: bandwidth - a row vector with the two optimal
25 % bandwidths for a bivaroate Gaussian kernel;
26 % the format is:
27 % bandwidth=[bandwidth_X, bandwidth_Y];
28 % density - an n by n matrix containing the density values over the n by n grid;
29 % density is not computed unless the function is asked for such an output;
30 % X,Y - the meshgrid over which the variable "density" has been computed;
31 % the intended usage is as follows:
32 % surf(X,Y,density)
33 % Example (simple Gaussian mixture)
34 % clear all
35 % % generate a Gaussian mixture with distant modes
36 % data=[randn(500,2);
37 % randn(500,1)+3.5, randn(500,1);];
38 % % call the routine
39 % [bandwidth,density,X,Y]=kde2d(data);
40 % % plot the data and the density estimate
41 % contour3(X,Y,density,50), hold on
42 % plot(data(:,1),data(:,2),'r.','MarkerSize',5)
43 %
44 % Example (Gaussian mixture with distant modes):
45 %
46 % clear all
47 % % generate a Gaussian mixture with distant modes
48 % data=[randn(100,1), randn(100,1)/4;
49 % randn(100,1)+18, randn(100,1);
50 % randn(100,1)+15, randn(100,1)/2-18;];

106 global N A2 I
107 if nargin<2
108 n=2^8;
109 end
110 n=2^ceil(log2(n)); % round up n to the next power of 2;
111 N=size(data,1);
112 if nargin<3
113 MAX=max(data,[],1); MIN=min(data,[],1); Range=MAX-MIN;
114 MAX_XY=MAX+Range/2; MIN_XY=MIN-Range/2;
115 end
116 scaling=MAX_XY-MIN_XY;
117 if N<=size(data,2)
118 error('data has to be an N by 2 array where each row represents a two dimensional observation')
119 end
120 transformed_data=(data-repmat(MIN_XY,N,1))./repmat(scaling,N,1);
121 %bin the data uniformly using regular grid;
122 initial_data=ndhist(transformed_data,n);
123 % discrete cosine transform of initial data
124 a= dct2d(initial_data);
125 % now compute the optimal bandwidth^2
126 I=(0:n-1).^2; A2=a.^2;
127 t_star=root(@(t)(t-evolve(t)),N);
128 p_02=func([0,2],t_star);p_20=func([2,0],t_star); p_11=func([1,1],t_star);
129 t_y=(p_02^(3/4)/(4*pi*N*p_20^(3/4)*(p_11+sqrt(p_20*p_02))))^(1/3);
130 t_x=(p_20^(3/4)/(4*pi*N*p_02^(3/4)*(p_11+sqrt(p_20*p_02))))^(1/3);
131 % smooth the discrete cosine transform of initial data using t_star
132 a_t=exp(-(0:n-1)'.^2*pi^2*t_x/2)*exp(-(0:n-1).^2*pi^2*t_y/2).*a;
133 % now apply the inverse discrete cosine transform
134 if nargout>1
135 density=idct2d(a_t)*(numel(a_t)/prod(scaling));
136 density(density<0)=eps; % remove any negative density values
137 [X,Y]=meshgrid(MIN_XY(1):scaling(1)/(n-1):MAX_XY(1),MIN_XY(2):scaling(2)/(n-1):MAX_XY(2));
138 end
139 bandwidth=sqrt([t_x,t_y]).*scaling;
140end

144function [out,time]=evolve(t)
145 global N
146 Sum_func = func([0,2],t) + func([2,0],t) + 2*func([1,1],t);
147 time=(2*pi*N*Sum_func)^(-1/3);
148 out=(t-time)/time;
149end

166function out=psi(s,Time)
167 global I A2
168 % s is a vector
169 w=exp(-I*pi^2*Time).*[1,.5*ones(1,length(I)-1)];
170 wx=w.*(I.^s(1));
171 wy=w.*(I.^s(2));
172 out=(-1)^sum(s)*(wy*A2*wx')*pi^(2*sum(s));
173end

183function data=dct2d(data)
184 % computes the 2 dimensional discrete cosine transform of data
185 % data is an nd cube
186 [nrows,ncols]= size(data);
187 if nrows~=ncols
188 error('data is not a square array!')
189 end
190 % Compute weights to multiply DFT coefficients
191 w = [1;2*(exp(-i*(1:nrows-1)*pi/(2*nrows))).'];
192 weight=w(:,ones(1,ncols));
193 data=dct1d(dct1d(data)')';
194 function transform1d=dct1d(x)
195
196 % Re-order the elements of the columns of x
197 x = [ x(1:2:end,:); x(end:-2:2,:) ];
198
199 % Multiply FFT by weights:
200 transform1d = real(weight.* fft(x));
201 end
202end

206function data = idct2d(data)
207 % computes the 2 dimensional inverse discrete cosine transform
208 [nrows,ncols]=size(data);
209 % Compute wieghts
210 w = exp(i*(0:nrows-1)*pi/(2*nrows)).';
211 weights=w(:,ones(1,ncols));
212 data=idct1d(idct1d(data)');
213 function out=idct1d(x)
214 y = real(ifft(weights.*x));
215 out = zeros(nrows,ncols);
216 out(1:2:nrows,:) = y(1:nrows/2,:);
217 out(2:2:nrows,:) = y(nrows:-1:nrows/2+1,:);
218 end
219end

223function binned_data=ndhist(data,M)
224 % this function computes the histogram
225 % of an n-dimensional data set;
226 % 'data' is nrows by n columns
227 % M is the number of bins used in each dimension
228 % so that 'binned_data' is a hypercube with
229 % size length equal to M;
230 [nrows,ncols]=size(data);
231 bins=zeros(nrows,ncols);
232 for i=1:ncols
233 [dum,bins(:,i)] = histc(data(:,i),[0:1/M:1],1);
234 bins(:,i) = min(bins(:,i),M);
235 end
236 % Combine the vectors of 1D bin counts into a grid of nD bin
237 % counts.
238 binned_data = accumarray(bins(all(bins>0,2),:),1/nrows,M(ones(1,ncols)));
239end

243function t=root(f,N)
244 % try to find smallest root whenever there is more than one
245 N=50*(N<=50)+1050*(N>=1050)+N*((N<1050)&(N>50));
246 tol=10^-12+0.01*(N-50)/1000;
247 flag=0;
248 while flag==0
249 try
250 t=fzero(f,[0,tol]);
251 flag=1;
252 catch
253 tol=min(tol*2,.1); % double search interval
254 end
255 if tol==.1 % if all else fails
256 t=fminbnd(@(x)abs(f(x)),0,.1); flag=1;
257 end
258 end
