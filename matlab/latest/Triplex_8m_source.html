1%> \brief
2%> This is the base class for generating instances of
3%> figures containing a square symmetric tiling of subplots.<br>
4%>
5%> \details
6%> This class generates figures containing three types of plots in the
7%> upper-triangle, lower-triangle, and diagonal subplots of the figure.<br>
8%> The dataset is assumed to be common among all plots (though not necessarily).<br>
9%> As such, all axes labels and tick marks of all subplots are dropped by default,
10%> except for subplots in the the left and bottom boundaries of the figure.<br>
11%>
12%> If any of the three input subplot types has colorbar, it is disabled for individual subplots.<br>
13%> Instead, a universal colorbar will be visualized on the right side of the Triplex plot.<br>
14%>
15%> \note
16%> See the list of class attributes below,
17%> also those of the superclass [pm.vis.figure.Figure](@ref Figure).<br>

553 function setLayout(self)
554 self.axes.height = 1 - self.axes.margin.top;
555 self.axes.width = 1 - self.axes.margin.right;
556 self.axes.subplot.width = (1 - self.axes.margin.left - self.axes.margin.right) / size(self.subplot, 1) - self.axes.subplot.interspacex;
557 self.axes.subplot.height = (1 - self.axes.margin.top - self.axes.margin.bottom) / size(self.subplot, 2) - self.axes.subplot.interspacey;
558 self.colorbar.origin.x = 1 - self.axes.margin.right;
559 self.colorbar.origin.y = self.axes.margin.bottom;
560 self.colorbar.height = size(self.subplot, 1) * (self.axes.subplot.height + self.axes.subplot.interspacey) - self.axes.subplot.interspacex;
561 self.figure.width = self.figure.height * (1 ...
562 - self.axes.margin.bottom ...
563 + self.axes.margin.right ...
564 + self.axes.margin.left ...
565 - self.axes.margin.top ...
566 + self.colorbar.width ...
567 );
568 if isfield(self, "fout") && isfield(self.fout, "gca") && isgraphics(self.fout.axes)
569 set(self.fout.axes, "position", self.getAxesPositionMain());
570 end
571 if any(strcmp(fieldnames(self.fout), "colorbar")) && isgraphics(self.fout.colorbar)
572 set(self.fout.colorbar, "position", self.getColorbarAxisPosition());
573 end

575 for icol = 1 : size(self.subplot, 2)
576 jrow = size(self.subplot, 1) - irow + 1;
577 if isgraphics(self.fout.subplotList{irow, icol}.fout.axes)
