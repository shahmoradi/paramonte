1847 subroutine getDistSortedExpDiffWeightKS2(SortedIndex, disSortedExpDiff, logVolUnitBall, ProbKS, StatKS, Lambda, alpha, beta, Err)
1848#if __INTEL_COMPILER && DLL_ENABLED && (WINDOWS_ENABLED || DARWIN_ENABLED)
1849 !DEC$ ATTRIBUTES DLLEXPORT :: getDistSortedExpDiffWeightKS2
1850#endif
1851 use pm_interpLinear, only: InterpLinear_type ! LCOV_EXCL_LINE
1852 use pm_statest, only: setSampleDisparityAuto ! LCOV_EXCL_LINE
1853 use Set_pmod, only: complementOrdered ! LCOV_EXCL_LINE
1854 use pm_kind, only: IK, RK ! LCOV_EXCL_LINE
1855 use pm_arraySort, only: setSorted ! LCOV_EXCL_LINE
1856 use pm_err, only: err_type ! LCOV_EXCL_LINE
1857 implicit none
1858 integer(IK) , intent(in) , contiguous :: SortedIndex(:)
1859 real(RK) , intent(in) , contiguous :: disSortedExpDiff(:)
1860 real(RK) , intent(in) , optional :: logVolUnitBall
1861 real(RK) , intent(out) , optional :: ProbKS(size(disSortedExpDiff))
1862 real(RK) , intent(out) , optional :: StatKS(size(disSortedExpDiff))
1863 real(RK) , intent(out) , optional :: Lambda(size(disSortedExpDiff))
1864 real(RK) , intent(out) , optional :: alpha, beta
1865 type(err_type) , intent(out) , optional :: Err
1866 real(RK) :: SortedIndexProbKS(size(SortedIndex))
1867 real(RK) :: SortedIndexStatKS(size(SortedIndex))
1868 real(RK) :: SortedIndexLambda(size(SortedIndex))
1869 integer(IK) :: IndexMap(size(disSortedExpDiff))
1870 real(RK) :: IndexAll(size(disSortedExpDiff))
1871 real(RK) , allocatable :: InterpolationPrediction(:)
1872 integer(IK) , allocatable :: InterpolationIndex(:)
1873 type(InterpLinear_type) :: Interp
1874 if (present(ProbKS) .or. present(StatKS) .or. present(Lambda) .or. present(alpha) .or. present(beta)) then
1875 call setSampleDisparityAuto( SampleSize = SortedIndex & ! LCOV_EXCL_LINE
1876 , SortedSample = disSortedExpDiff & ! LCOV_EXCL_LINE
1877 , ProbKS = SortedIndexProbKS & ! LCOV_EXCL_LINE
1878 , StatKS = SortedIndexStatKS & ! LCOV_EXCL_LINE
1879 , Lambda = SortedIndexLambda & ! LCOV_EXCL_LINE
1880 , Err = Err & ! LCOV_EXCL_LINE
1881 )
1882 if (err%occurred) return
1883 if (present(alpha)) alpha = sum(SortedIndexProbKS)
1884 if (present(beta)) beta = sum(SortedIndexProbKS * disSortedExpDiff(SortedIndex))
1885 blockFullSortedIndex: if (size(SortedIndex) < size(disSortedExpDiff)) then ! Interpolate the missing weights.
1886 InterpolationIndex = complementOrdered(SortedIndex, lower = 1_IK, upper = size(disSortedExpDiff))
1887 Interp = InterpLinear_type(SortedX = real(SortedIndex,RK), SortedY = SortedIndexProbKS, Err = Err)
1888 if (present(Err)) then; if (err%occurred) return; end if ! LCOV_EXCL_LINE
1889 InterpolationPrediction = Interp%predict(QueryX = real(InterpolationIndex,RK))
1890 if (present(beta)) beta = beta + sum(InterpolationPrediction * disSortedExpDiff(InterpolationIndex))
1891 if (present(alpha)) alpha = alpha + sum(InterpolationPrediction)
1892 if (present(ProbKS) .or. present(StatKS) .or. present(Lambda)) then
1893 IndexAll = [SortedIndex, InterpolationIndex]
1894 call setSorted(IndexAll, IndexMap)
1895 IndexAll = IndexAll(IndexMap)
1896 if (present(ProbKS)) then
1897 ProbKS = [SortedIndexProbKS, InterpolationPrediction]
1898 ProbKS = ProbKS(IndexMap)
1899 end if
1900 if (present(StatKS)) then
1901 Interp = InterpLinear_type(SortedX = real(SortedIndex,RK), SortedY = SortedIndexStatKS, Err = Err)
1902 if (present(Err)) then; if (err%occurred) return; end if ! LCOV_EXCL_LINE
1903 InterpolationPrediction = Interp%predict(QueryX = real(InterpolationIndex,RK))
1904 StatKS = [SortedIndexStatKS, InterpolationPrediction]
1905 StatKS = StatKS(IndexMap)
1906 end if
1907 if (present(Lambda)) then
1908 Interp = InterpLinear_type(SortedX = real(SortedIndex,RK), SortedY = SortedIndexLambda, Err = Err)
1909 if (present(Err)) then; if (err%occurred) return; end if ! LCOV_EXCL_LINE
1910 InterpolationPrediction = Interp%predict(QueryX = real(InterpolationIndex,RK))
1911 Lambda = [SortedIndexLambda, InterpolationPrediction]
1912 Lambda = Lambda(IndexMap)
1913 end if
1914 end if
1915#if CHECK_ENABLED
1916 block
1917 use iso_fortran_env, only: output_unit
1918 integer(IK) :: ip, npMissing, counterPredictionIndex
1919 ! Check the consistency of the indices
1920 do ip = 1, size(SortedIndex)
1921 if (any(InterpolationIndex == SortedIndex(ip))) then
1922 write(output_unit,"(*(g0:,', '))") "InterpolationIndex == SortedIndex for ip, SortedIndex(ip)", ip, SortedIndex(ip)
1923 error stop
1924 end if
1925 end do
1926 npMissing = size(disSortedExpDiff) - size(SortedIndex)
1927 counterPredictionIndex = size(InterpolationIndex, kind = IK)
1928 if (counterPredictionIndex /= npMissing) then
1929 write(output_unit,"(*(g0:,', '))") "counterPredictionIndex /= npMissing", counterPredictionIndex, npMissing
1930 error stop
1931 end if
1932 end block
1933#endif
1934 deallocate(InterpolationIndex, InterpolationPrediction)
1935 else blockFullSortedIndex
1936 if (present(ProbKS)) ProbKS = SortedIndexProbKS
1937 if (present(StatKS)) StatKS = SortedIndexStatKS
1938 if (present(Lambda)) Lambda = SortedIndexLambda
1939 end if blockFullSortedIndex
1940 if (present(logVolUnitBall) .and. present(beta)) beta = exp(log(beta) + logVolUnitBall)
1941 else
1942 if (present(Err)) err%occurred = .false._LK
1943 end if
1944 end subroutine getDistSortedExpDiffWeightKS2
